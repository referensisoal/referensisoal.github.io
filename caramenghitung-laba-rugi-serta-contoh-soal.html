1 pemuaian panjang = [tex]L_T [/tex] =[tex]L_O[/tex] ( 1 + [tex] \alpha [/tex] . Δt ) 
 Keterangan : [tex]L_t[/tex] =panjang benda setelah memuai 
                     [tex]L_O[/tex] =panjang benda mula mula 
                     α                    = koefisien muai panjang 
2 pemuaian luas  = [tex]A_t [/tex]= [tex]A_o [/tex] ( 1 + β . Δt ) 
keterangan : [tex]A_t[/tex] = luas benda setelah memuai 
                   [tex]A_O[/tex] = luas benda mula mula
                   β                   = koefisien muai luas = 2α
                   Δt                  = kenaikan suhu ( jika suhu nya ada dua maka . Δt 2 - Δt 1 )
3 pemuaian volume = [tex]V_t[/tex] = [tex]V_o[/tex] ( 1+gama .Δt )
keterangan ; [tex]V_t[/tex] = volume benda setelah memuai 
                   [tex]V_o[/tex] = volume benda mula mula 
                   gama             = koefisien muai volume = 3α
4 pemuaian gas = [tex]V_t[/tex] = [tex]V_o [/tex] (1 + Δt/273)
keterangan : [tex]V_t[/tex] = volume gas setelah memuai
